export function WriteWorld() {
  return <>World</>
}

export function AsList(items) {
  return (
    <ul>
      {items.map(item => (
        <li key={item.id}>{item.name}</li>
      ))}
    </ul>
  )
}

export function AsMarkdownList(items) {
  return (<>
    {items.map(item => (
      `- ${item.name}`
    ))}
  </>
  )
}
